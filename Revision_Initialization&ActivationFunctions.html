class Reg_FFN:
  def __init__(self, initialization = 'random', activation = 'tanh', leaky_slop = 0.1):
    self.params = {}
    self.pramas_h = []
    self.gradients = {}
    self.update_params = {}
    self.prev_update_params = {}
    self.no_layers = 3
    self.layers = [2, 2, 3, 4]
    self.activation = activation
    self.leaky_slop = leaky_slop

    if initialization == 'zero':
      for i in range(1, self.no_layers+1):
        self.params['W'+str(i)] = np.zeros((self.layers[i-1], self.layers[i]))
        self.params['B'+str(i)] = np.zeros((1, self.layers[i]))
    
    elif initialization == 'random':
      for i in range(1, self.no_layers+1):
        self.params['W'+str(i)] = np.random.randn(self.layers[i-1], self.layers[i])
        self.params['B'+str(i)] = np.random.randn(1, self.layers[i])
    
    elif initialization == 'HE':
      for i in range(1, self.no_layers+1):
        self.params['W'+str(i)] = np.random.randn(self.layers[i-1], self.layers[i]) * np.sqrt(2 / self.layers[i-1])
        self.params['B'+str(i)] = np.random.randn(1, self.layers[i])
    
    elif initialization == 'xavier':
      for i in range(1, self.no_layers+1):
        self.params['W'+str(i)] = np.random.randn(self.layers[i-1], self.layers[i]) * np.sqrt(1 / self.layers[i-1])
        self.params['B'+str(i)] = np.random.randn(1, self.layers[i])
    
    for i in range(1, self.no_layers+1):
      self.update_params["v_w"+str(i)] = 0
      self.update_params["v_b"+str(i)] = 0
      self.update_params["m_w"+str(i)] = 0
      self.update_params["m_b"+str(i)] = 0
      self.prev_update_params["v_w"+str(i)] = 0
      self.prev_update_params["v_b"+str(i)] = 0
  
  def forward_activation(self, X):
    if self.activation == 'sigmoid':
      return 1 / (1 + np.exp(-X))
    
    elif self.activation == 'tanh':
      return np.tanh(X)
    
    elif self.activation == 'ReLu':
      return np.maximum(0, X)
    
    elif self.activation == 'Leaky_ReLu':
      return np.maximum(self.leaky_slop*X, X)
    
  def grad_activation(self, X):
    if self.activation == 'sigmoid':
      return X*(1-X)
    
    elif self.activation == 'tanh':
      return (1-np.square(X))
    
    elif self.activation == 'ReLu':
      return 1.0 * (X>0)
    
    elif self.activation == 'Leaky_ReLu':
      d = np.zeros_like(X)
      d[X>0] = 1
      d[X<=0] = self.leaky_slop
      return d
  
  def softmax(self, X):
    exps = np.exp(X)
    return exps / np.sum(exps, 1).reshape(-1, 1)

  def forward_pass(self, X, params = None):
    if params == None:
      params = self.params
    
    self.A1 = np.matmul(X, params['W1']) + params['B1']   # (N, 2) * (2, 2) + (1, 2) -> (N, 2)
    self.H1 = self.forward_activation(self.A1)
    self.A2 = np.matmul(self.H1, params['W2']) + params['B2']  # (N, 2) * (2, 3) + (1, 3) -> (N, 3)
    self.H2 = self.forward_activation(self.A2)
    self.A3 = np.matmul(self.H2, params['W3']) + params['B3']  # (N, 3) * (3, 4) + (1, 4) -> (N, 4)
    self.H3 = self.softmax(self.A3)
    return self.H3      # (N, 4)
  
  def grad(self, X, Y, params = None):
    if params == None:
      params = self.params
    
    self.forward_pass(X, params)

    self.gradients['dA3'] = self.H3 - Y       # (N, 4)
    self.gradients['dW3'] = np.matmul(self.H2.T, self.gradients['dA3'])     # (3, N) * (N, 4) -> (3, 4)
    self.gradients['dB3'] = np.sum(self.gradients['dA3'], 0).reshape(1, -1) # (1, 4)
    self.gradients['dH2'] = np.matmul(self.gradients['dA3'], self.params['W3'].T) # (N, 4) * (4, 3) -> (N, 3)
    self.gradients['dA2'] = np.multiply(self.gradients['dH2'], self.grad_activation(self.H2))   # (N, 3) * (N, 3) -> (N, 3)
    self.gradients['dW2'] = np.matmul(self.H1.T, self.gradients['dA2'])     # (2, N) * (N, 3) -> (2, 3)
    self.gradients['dB2'] = np.sum(self.gradients['dA2'], 0).reshape(1, -1) # (1, 3)
    self.gradients['dH1'] = np.matmul(self.gradients['dA2'], self.params['W2'].T) # (N, 3) * (3, 2) -> (N, 2)
    self.gradients['dA1'] = np.multiply(self.gradients['dH1'], self.grad_activation(self.H1))  # (N, 2) * (N, 2) -> (N, 2)
    self.gradients['dW1'] = np.matmul(X.T, self.gradients['dA1'])     # (2, N) * (N, 2) -> (2, 2)
    self.gradients['dB1'] = np.sum(self.gradients['dA1'], 0).reshape(1, -1) # (1, 2)
  
  def fit(self, X, Y, epochs = 10, algo = 'GD', mini_batch_size = 32, eps = 1e-8, lr = 0.05, beta = 0.9, beta1 = 0.9, beta2 = 0.9, gamma = 0.9):
    loss = {}
    for epoch in tqdm_notebook(range(epochs), total = epochs, unit = 'epoch'):
      m = X.shape[0]

      if algo == 'GD':
        self.grad(X, Y)
        for i in range(1, self.no_layers+1):
          self.params['W'+str(i)] -= self.gradients['dW'+str(i)] / m
          self.params['B'+str(i)] -= self.gradients['dB'+str(i)] / m
      
      elif algo == 'mini_batch':
        for k in range(0, m, mini_batch_size):
          self.grad(X[k:k+mini_batch_size], Y[k:k+mini_batch_size])
          for i in range(1, self.no_layers+1):
            self.params['W'+str(i)] -= lr * (self.gradients['dW'+str(i)] / mini_batch_size)
            self.params['B'+str(i)] -= lr * (self.gradients['dB'+str(i)] / mini_batch_size)
      
      elif algo == 'Momentum':
        self.grad(X, Y)
        for i in range(1, self.no_layers+1):
          self.update_params['v_w'+str(i)] = gamma * self.update_params['v_w'+str(i)] + lr * self.gradients['dW'+str(i)]
          self.update_params['v_b'+str(i)] = gamma * self.update_params['v_b'+str(i)] + lr * self.gradients['dB'+str(i)]
          self.params['W'+str(i)] -= self.update_params['v_w'+str(i)]
          self.params['B'+str(i)] -= self.update_params['v_b'+str(i)]
      
      elif algo == "NAG":
        temp_params = {}
        for i in range(1,self.no_layers+1):
          self.update_params["v_w"+str(i)]=gamma*self.prev_update_params["v_w"+str(i)]
          self.update_params["v_b"+str(i)]=gamma*self.prev_update_params["v_b"+str(i)]
          temp_params["W"+str(i)]=self.params["W"+str(i)]-self.update_params["v_w"+str(i)]
          temp_params["B"+str(i)]=self.params["B"+str(i)]-self.update_params["v_b"+str(i)]
        self.grad(X,Y,temp_params)
        for i in range(1,self.no_layers+1):
          self.update_params["v_w"+str(i)] = gamma *self.update_params["v_w"+str(i)] + lr * (self.gradients["dW"+str(i)]/m)
          self.update_params["v_b"+str(i)] = gamma *self.update_params["v_b"+str(i)] + lr * (self.gradients["dB"+str(i)]/m)
          self.params["W"+str(i)] -= lr * (self.update_params["v_w"+str(i)])
          self.params["B"+str(i)] -= lr * (self.update_params["v_b"+str(i)]) 
        self.prev_update_params=self.update_params 
        
      elif algo == "AdaGrad":
        self.grad(X, Y)
        for i in range(1,self.no_layers+1):
          self.update_params["v_w"+str(i)] += (self.gradients["dW"+str(i)]/m)**2
          self.update_params["v_b"+str(i)] += (self.gradients["dB"+str(i)]/m)**2
          self.params["W"+str(i)] -= (lr/(np.sqrt(self.update_params["v_w"+str(i)])+eps)) * (self.gradients["dW"+str(i)]/m)
          self.params["B"+str(i)] -= (lr/(np.sqrt(self.update_params["v_b"+str(i)])+eps)) * (self.gradients["dB"+str(i)]/m)
      
      elif algo == "RMSProp":
        self.grad(X, Y)
        for i in range(1,self.no_layers+1):
          self.update_params["v_w"+str(i)] = beta*self.update_params["v_w"+str(i)] +(1-beta)*((self.gradients["dW"+str(i)]/m)**2)
          self.update_params["v_b"+str(i)] = beta*self.update_params["v_b"+str(i)] +(1-beta)*((self.gradients["dB"+str(i)]/m)**2)
          self.params["W"+str(i)] -= (lr/(np.sqrt(self.update_params["v_w"+str(i)]+eps)))*(self.gradients["dW"+str(i)]/m)
          self.params["B"+str(i)] -= (lr/(np.sqrt(self.update_params["v_b"+str(i)]+eps)))*(self.gradients["dB"+str(i)]/m)
      
      elif algo == "Adam":
        self.grad(X, Y)
        num_updates=0
        for i in range(1,self.no_layers+1):
          num_updates+=1
          self.update_params["m_w"+str(i)]=beta1*self.update_params["m_w"+str(i)]+(1-beta1)*(self.gradients["dW"+str(i)]/m)
          self.update_params["v_w"+str(i)]=beta2*self.update_params["v_w"+str(i)]+(1-beta2)*((self.gradients["dW"+str(i)]/m)**2)
          m_w_hat=self.update_params["m_w"+str(i)]/(1-np.power(beta1,num_updates))
          v_w_hat=self.update_params["v_w"+str(i)]/(1-np.power(beta2,num_updates))
          self.params["W"+str(i)] -=(lr/np.sqrt(v_w_hat+eps))*m_w_hat
          
          self.update_params["m_b"+str(i)]=beta1*self.update_params["m_b"+str(i)]+(1-beta1)*(self.gradients["dB"+str(i)]/m)
          self.update_params["v_b"+str(i)]=beta2*self.update_params["v_b"+str(i)]+(1-beta2)*((self.gradients["dB"+str(i)]/m)**2)
          m_b_hat=self.update_params["m_b"+str(i)]/(1-np.power(beta1,num_updates))
          v_b_hat=self.update_params["v_b"+str(i)]/(1-np.power(beta2,num_updates))
          self.params["B"+str(i)] -=(lr/np.sqrt(v_b_hat+eps))*m_b_hat
      
      Y_pred = self.predict(X)
      loss[epoch] = log_loss(np.argmax(Y, 1), Y_pred)
    plt.plot(list(loss.values()))
    plt.xlabel('Epoch')
    plt.ylabel('Loss')
    plt.show()

  def predict(self, X):
    Y_pred = self.forward_pass(X)
    return np.array(Y_pred).squeeze()
